001package com.arcao.geocaching.api;
002
003import com.google.auto.value.AutoValue;
004
005import org.jetbrains.annotations.Nullable;
006
007import java.io.Serializable;
008
009@AutoValue
010public abstract class Status implements Serializable {
011    private static final long serialVersionUID = -6603595166996374235L;
012
013    public abstract int code();
014
015    public StatusCode statusCode() {
016        return StatusCode.getByCode(code());
017    }
018
019    public abstract String message();
020
021    @Nullable
022    public abstract String exceptionDetails();
023
024    public static Builder builder() {
025        return new AutoValue_Status.Builder();
026    }
027
028    @AutoValue.Builder
029    public abstract static class Builder {
030        public abstract Builder code(int code);
031
032        public abstract Builder message(String message);
033
034        public abstract Builder exceptionDetails(@Nullable String exceptionDetails);
035
036        public abstract Status build();
037    }
038}
