9   /**
10   * Container class for API limits related to currently logged user
11   *
12   * @author arcao
13   */
14  @AutoValue
15  public abstract class ApiLimits implements Serializable {
16      private static final long serialVersionUID = -3608995080972521881L;
17  
18      /**
19       * Returns limits for downloading caches
20       *
21       * @return cache limit object
22       */
23      public abstract List<CacheLimit> cacheLimits();
24  
25      /**
26       * Returns if the cache limits are enforced
27       *
28       * @return are enforced
29       */
30      public abstract boolean enforceCacheLimits();
31  
32      /**
33       * Returns if the cache limits are enforced for a lite caches
34       *
35       * @return are enforced
36       */
37      public abstract boolean enforceLiteCacheLimits();
38  
39      /**
40       * Returns if the method limits are enforced
41       *
42       * @return are enforced
43       */
44      public abstract boolean enforceMethodLimits();
45  
46      /**
47       * Returns a membership type for which are limits applied
48       *
49       * @return member type
50       */
51      public abstract MemberType forMembershipType();
52  
53      /**
54       * Returns a license key
55       *
56       * @return license key
57       */
58      public abstract String licenseKey();
59  
60      /**
61       * Returns a cache limits for lite caches
62       *
63       * @return cache limits
64       */
65      public abstract List<CacheLimit> liteCacheLimits();
66  
67      /**
68       * Returns the number of max calls to be performed with same IP per minute
69       *
70       * @return number of max calls
71       */
72      public abstract long maxCallsbyIPIn1Minute();
73  
74      /**
75       * Returns the method limits
76       *
77       * @return method limits
78       */
79      public abstract List<MethodLimit> methodLimits();
80  
81      /**
82       * Returns whereas limits are restricted by IP
83       *
84       * @return are restricted by IP
85       */
86      public abstract boolean restrictByIp();
87  
88      /**
89       * Returns if counts of IP are validated
90       *
91       * @return count of IP validated
92       */
93      public abstract boolean validateIpCounts();
94  
95      public static Builder builder() {
96          return new AutoValue_ApiLimits.Builder();
97      }
98  
99      @AutoValue.Builder
100     public abstract static class Builder {
101         public abstract Builder cacheLimits(List<CacheLimit> cacheLimits);
102         public abstract Builder enforceCacheLimits(boolean enforceCacheLimits);
103         public abstract Builder enforceLiteCacheLimits(boolean enforceLiteCacheLimits);
104         public abstract Builder enforceMethodLimits(boolean enforceMethodLimits);
105         public abstract Builder forMembershipType(MemberType forMembershipType);
106         public abstract Builder licenseKey(String licenseKey);
107         public abstract Builder liteCacheLimits(List<CacheLimit> liteCacheLimits);
108         public abstract Builder maxCallsbyIPIn1Minute(long maxCallsbyIPIn1Minute);
109         public abstract Builder methodLimits(List<MethodLimit> methodLimits);
110         public abstract Builder restrictByIp(boolean restrictByIp);
111         public abstract Builder validateIpCounts(boolean validateIpCounts);
112         public abstract ApiLimits build();
113     }
