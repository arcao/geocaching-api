25      /**

28       * @param id Groundspeak Id
29       * @return Member Type
30       */
31      @NotNull
32      public static MemberType fromId(int id) {
33          for (MemberType memberType : values()) {
34              if (memberType.id == id)
35                  return memberType;
36          }
37  
38          return Guest;
39      }
