1   package com.arcao.geocaching.api.filter;
2   
3   import com.arcao.geocaching.api.data.coordinates.Coordinates;
4   import com.google.gson.stream.JsonWriter;
5   
6   import org.jetbrains.annotations.NotNull;
7   
8   import java.io.IOException;
9   
10  public class PointRadiusFilter implements Filter {
11      private static final String NAME = "PointRadius";
12  
13      private final long distanceInMeters;
14      private final double latitude;
15      private final double longitude;
16  
17      public PointRadiusFilter(double latitude, double longitude, long distanceInMeters) {
18          this.latitude = latitude;
19          this.longitude = longitude;
20          this.distanceInMeters = distanceInMeters;
21      }
22  
23      /**
24       * @since 1.5
25       */
26      public PointRadiusFilter(@NotNull Coordinates coordinates, long distanceInMeters) {
27          this(coordinates.latitude(), coordinates.longitude(), distanceInMeters);
28      }
29  
30      @NotNull
31      @Override
32      public String name() {
33          return NAME;
34      }
35  
36      @Override
37      public boolean valid() {
38          return true;
39      }
40  
41      @Override
42      public void writeJson(@NotNull JsonWriter w) throws IOException {
43          w.name(NAME);
44          w.beginObject();
45  
46          w.name("DistanceInMeters").value(distanceInMeters);
47  
48          w.name("Point");
49          w.beginObject();
50          w.name("Latitude").value(latitude);
51          w.name("Longitude").value(longitude);
52          w.endObject();
53  
54          w.endObject();
55      }
56  }
