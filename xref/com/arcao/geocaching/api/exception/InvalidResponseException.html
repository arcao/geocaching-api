13          super(String.valueOf(statusCode) + " " + statusMessage + "\n\n" + data);
14          this.statusCode = statusCode;
15          this.statusMessage = statusMessage;
16          this.data = data;
17      }
18  
19      public InvalidResponseException(String message, Throwable cause) {
