8   /**
9    * Basic interface for configuration class

13  public interface GeocachingApiConfiguration {

15       * Production configuration
16       */

18      /**
19       * Staging configuration
20       */
21      GeocachingApiConfiguration STAGING = new DefaultStagingGeocachingApiConfiguration();
22  
23      /**
24       * Returns a common part of Gecaching API service URL
25       *
26       * @return url
27       */
28      @NotNull
29      String getApiServiceEntryPointUrl();
30  
31      /**
32       * Sets a specified timeout value, in milliseconds, to be used when opening
33       * a communications link. A timeout of zero is interpreted as an infinite timeout.
34       *
35       * @return an <code>int</code> that specifies the connect timeout value in milliseconds
36       * @since 1.5.16
37       */
38      int getConnectTimeout();
39  
40      /**
41       * Sets the read timeout to a specified timeout, in milliseconds. A non-zero
42       * value specifies the timeout when reading from Input stream when a connection is
43       * established to a resource. A timeout of zero is interpreted as an infinite timeout.
44       *
45       * @return an <code>int</code> that specifies the timeout value to be used in milliseconds
46       * @since 1.5.16
47       */
48      int getReadTimeout();
49  }
